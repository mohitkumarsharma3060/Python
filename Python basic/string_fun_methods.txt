upper()->	make all upper
title()->	make first char of each word capital ,rest lower
swapcase()->	swap the case of each char
format()->	use to format a string
index()->	lowest index of argumneted char
count()->	frequency of argumented char
startswith()->  true if calling string starts with arg str
endswith()->	true if calling string ends with arg str
